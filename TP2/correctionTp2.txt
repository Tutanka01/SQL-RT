5. noms des machines equipees d'un processeur supportant les instructions 3D Now
select distinct m.nompc from machine m, composant c, assemble a
    ...> INNER JOIN assemble ON a.numPC = m.numpc
    ...> INNER JOIN assemble ON a.refcomp = c.refcomp
    ...> where c.type like '%Processeur%' and c.nomcomp like '%3D%';
